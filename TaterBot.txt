//Use with rectangular carrot farm
//Set up farm with hoppers at one x end in a row leading to a chest
//Configure inventory with a fortune 3 diamond pick in slot one and a stack of carrots in slot 9 (for a buffer)
//Make rows no longer than 150 blocks

SET(#rows,?) //set ? to the number of rows of your farm (including empty rows if you have any)
DO
TOGGLEKEY(FORWARD)
DO(%#rows%)
DO
SLOT(1)
LOOK(90,90)
KEY(ATTACK)
UNTIL(XPOS=?) //Set ? to the far x end of your farm
DO
PICK(potato)
LOOK(270,90)
KEY(USE)
UNTIL(XPOS=-6448)
TOGGLEKEY(FORWARD)
KEYDOWN(SHIFT)
SET(#slot,2)
DO
LOOK(270,60)
SLOT(%#slot%)
GetSlotItem(%#slot%,#id,#num)
DO(%#num%)
KEY(DROP)
wait(10ms)
LOOP
INC(#slot,1)
UNTIL(#slot=9)
KEYUP(SHIFT)
TOGGLEKEY(FORWARD)
LOOK(0,0)
WAIT(4t)
LOOP
DO
LOOK(180,0)
UNTIL(XPOS=?) //Set ? to the hopper end of the farm
TOGGLEKEY(FORWARD)
WAIT(?t) //Replace ? with time to wait for crops to grow (hours*72000)
LOOP
